## Schema Completo

```sql
-- ===========================
-- EDITAIS (M1 - Radar)
-- ===========================

CREATE TABLE edital_sources (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,              -- "Compras.gov.br", "BEC-SP"
  portal_url TEXT,
  scraper_type TEXT,               -- "api", "headless", "hybrid"
  is_active BOOLEAN DEFAULT true,
  last_scraped_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE editals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  source_id UUID REFERENCES edital_sources(id),
  external_id TEXT,                -- ID no portal de origem
  portal_url TEXT,                 -- Link direto pro edital
  numero TEXT,                     -- Numero do pregao
  uasg TEXT,                       -- Codigo do orgao
  orgao TEXT NOT NULL,             -- Nome do orgao
  objeto TEXT NOT NULL,            -- Descricao do objeto
  modalidade TEXT,                 -- "pregao_eletronico", "concorrencia", etc.
  valor_estimado DECIMAL(15,2),
  local_entrega TEXT,
  prazo_entrega TEXT,
  data_sessao TIMESTAMPTZ,
  data_publicacao TIMESTAMPTZ,
  status TEXT DEFAULT 'novo' CHECK (status IN (
    'novo', 'analisando', 'go', 'no_go', 'proposta_enviada',
    'em_pregao', 'adjudicado', 'nao_adjudicado', 'contrato', 'descartado'
  )),
  pdf_url TEXT,
  pdf_storage_path TEXT,
  raw_data JSONB,                  -- Dados brutos do scraping
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE edital_filters (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  name TEXT NOT NULL,
  keywords TEXT[],                 -- ["cesta basica", "kit alimentacao"]
  regioes TEXT[],                  -- ["SP", "RJ", "MG"]
  valor_min DECIMAL(15,2),
  valor_max DECIMAL(15,2),
  modalidades TEXT[],
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- ===========================
-- ANALISE GO/NO-GO (M2)
-- ===========================

CREATE TABLE edital_analyses (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  edital_id UUID REFERENCES editals(id) ON DELETE CASCADE NOT NULL,
  analyzed_by UUID REFERENCES profiles(id),
  decision TEXT CHECK (decision IN ('go', 'no_go', 'pendente')),
  score_total DECIMAL(4,2),
  justificativa TEXT,
  criteria JSONB NOT NULL,         -- Array de {criterio, tipo, resultado, nota, observacao}
  analyzed_at TIMESTAMPTZ DEFAULT now(),
  created_at TIMESTAMPTZ DEFAULT now()
);

-- ===========================
-- DOCUMENTOS (M3)
-- ===========================

CREATE TABLE document_categories (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,              -- "Habilitacao Juridica", "Fiscal", etc.
  slug TEXT UNIQUE NOT NULL,
  sort_order INTEGER DEFAULT 0,
  is_system BOOLEAN DEFAULT false  -- Categorias padrao do sistema
);

CREATE TABLE documents (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  category_id UUID REFERENCES document_categories(id),
  name TEXT NOT NULL,              -- "CND Federal", "Alvara Sanitario"
  description TEXT,
  file_path TEXT,                  -- Path no Supabase Storage
  file_type TEXT,                  -- "pdf", "jpg", "png"
  file_size INTEGER,
  issued_at DATE,                  -- Data de emissao
  expires_at DATE,                 -- Data de vencimento
  status TEXT DEFAULT 'pendente' CHECK (status IN (
    'valido', 'vencendo', 'vencido', 'pendente'
  )),
  alert_days_before INTEGER DEFAULT 30,
  notes TEXT,
  version INTEGER DEFAULT 1,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE document_alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  document_id UUID REFERENCES documents(id) ON DELETE CASCADE NOT NULL,
  alert_date DATE NOT NULL,
  alert_type TEXT CHECK (alert_type IN ('vencendo', 'vencido', 'renovar')),
  sent BOOLEAN DEFAULT false,
  sent_at TIMESTAMPTZ,
  channel TEXT,                    -- "push", "email"
  created_at TIMESTAMPTZ DEFAULT now()
);

-- ===========================
-- CUSTO E PROPOSTA (M4)
-- ===========================

CREATE TABLE cost_sheets (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  edital_id UUID REFERENCES editals(id) ON DELETE SET NULL,
  name TEXT NOT NULL,              -- "Cesta Basica Padrao", "Kit PNAE"
  custo_total DECIMAL(15,2),
  margem_desejada DECIMAL(5,2),   -- % desejada
  margem_piso DECIMAL(5,2),       -- % minima (piso)
  preco_venda DECIMAL(15,2),
  preco_piso DECIMAL(15,2),       -- Preco no piso de margem
  preco_referencia DECIMAL(15,2), -- Preco do edital
  impostos_percentual DECIMAL(5,2),
  regime_tributario TEXT,          -- "simples", "presumido", "real"
  is_template BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE cost_items (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  cost_sheet_id UUID REFERENCES cost_sheets(id) ON DELETE CASCADE NOT NULL,
  categoria TEXT NOT NULL,         -- "materia_prima", "mao_de_obra", "embalagem", "frete"
  descricao TEXT NOT NULL,         -- "Arroz 5kg", "Feijao 1kg"
  quantidade DECIMAL(10,3),
  unidade TEXT,                    -- "kg", "un", "L"
  custo_unitario DECIMAL(10,2),
  custo_total DECIMAL(10,2),
  sort_order INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT now()
);

-- ===========================
-- PREGAO (M5)
-- ===========================

CREATE TABLE bidding_sessions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  edital_id UUID REFERENCES editals(id) ON DELETE CASCADE NOT NULL,
  cost_sheet_id UUID REFERENCES cost_sheets(id) ON DELETE SET NULL,
  data_sessao TIMESTAMPTZ NOT NULL,
  preco_piso DECIMAL(15,2),       -- Copiado da planilha
  preco_abertura DECIMAL(15,2),   -- Preco inicial ofertado
  preco_final DECIMAL(15,2),      -- Ultimo lance
  margem_final DECIMAL(5,2),      -- Margem do lance final
  resultado TEXT CHECK (resultado IN (
    'adjudicado', 'classificado', 'nao_classificado',
    'desclassificado', 'deserto', 'fracassado', 'pendente'
  )),
  classificacao INTEGER,           -- Posicao final (1o, 2o, 3o...)
  observacoes TEXT,
  pre_checklist JSONB,             -- Checklist pre-sessao preenchido
  created_at TIMESTAMPTZ DEFAULT now(),
  updated_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE bids (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id UUID REFERENCES bidding_sessions(id) ON DELETE CASCADE NOT NULL,
  valor DECIMAL(15,2) NOT NULL,
  margem DECIMAL(5,2),
  is_mine BOOLEAN DEFAULT true,    -- Meu lance ou do concorrente
  concorrente TEXT,                -- Nome/CNPJ do concorrente (se visivel)
  timestamp TIMESTAMPTZ DEFAULT now()
);

-- ===========================
-- NOTIFICACOES (M9)
-- ===========================

CREATE TABLE notifications (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  type TEXT NOT NULL,              -- "edital_novo", "certidao_vencendo", "pregao_amanha", etc.
  title TEXT NOT NULL,
  body TEXT,
  data JSONB,                      -- Payload (edital_id, document_id, etc.)
  read BOOLEAN DEFAULT false,
  read_at TIMESTAMPTZ,
  channel TEXT,                    -- "push", "email", "in_app"
  sent BOOLEAN DEFAULT false,
  sent_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE notification_preferences (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE NOT NULL,
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE NOT NULL,
  type TEXT NOT NULL,              -- "edital_novo", "certidao_vencendo", etc.
  push_enabled BOOLEAN DEFAULT true,
  email_enabled BOOLEAN DEFAULT true,
  UNIQUE(user_id, type)
);

-- ===========================
-- INDICES
-- ===========================

CREATE INDEX idx_editals_tenant ON editals(tenant_id);
CREATE INDEX idx_editals_status ON editals(tenant_id, status);
CREATE INDEX idx_editals_data_sessao ON editals(data_sessao);
CREATE INDEX idx_editals_objeto ON editals USING gin(to_tsvector('portuguese', objeto));
CREATE INDEX idx_documents_tenant ON documents(tenant_id);
CREATE INDEX idx_documents_expires ON documents(tenant_id, expires_at);
CREATE INDEX idx_notifications_user ON notifications(user_id, read);
CREATE INDEX idx_cost_sheets_tenant ON cost_sheets(tenant_id);
CREATE INDEX idx_bidding_sessions_tenant ON bidding_sessions(tenant_id);
```

## RLS (todas as tabelas)

Padrao: `tenant_id IN (SELECT tenant_id FROM profiles WHERE id = auth.uid())`

Excecoes:
- `edital_sources`: leitura publica (compartilhada entre tenants)
- `document_categories`: leitura publica (categorias do sistema)
